<p>you need hashCode and equals methods if you want to use your class as a key. let the be generated by your IDE or use:</p>

<pre><code>@Override
public int hashCode() {
    final int prime = 31;
    int result = 1;
    result = prime * result + age;
    result = prime * result + ( ( name == null ) ? 0 : name.hashCode() );
    return result;
}
@Override
public boolean equals( Object obj ) {
    if( this == obj )
        return true;
    if( obj == null )
        return false;
    if( getClass() != obj.getClass() )
        return false;
    Student2 other = (Student2) obj;
    if( age != other.age )
        return false;
    if( name == null ) {
        if( other.name != null )
            return false;
    } else if( !name.equals( other.name ) )
        return false;
    return true;
}
